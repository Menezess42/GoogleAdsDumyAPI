query
  = SELECT select_list
    FROM resource
    where_clause?
    order_by_clause?
    limit_clause? ;

select_list
  = field ( COMMA field )* ;

field
  = identifier DOT identifier ;

resource
  = identifier ;

where_clause
  = WHERE condition ( AND condition )* ;

condition
  = comparison
  | between_condition ;

comparison
  = field comparison_operator value ;

comparison_operator
  = IQUAL
  | MORE
  | LESS ;

between_condition
  = field BETWEEN value AND value ;

order_by_clause
  = ORDER_BY order_list ;

order_list
  = order_item ( COMMA order_item )* ;

order_item
  = field ( ASC )? ;

limit_clause
  = LIMIT number ;

value
  = string
  | number ;

identifier
  = letter ( letter | number | "_" )* ;

number
  = digit+ ( DOT digit+ )? ;

string
  = APOSTROPHE string_char* APOSTROPHE ;

string_char
  = letter
  | number
  | DASH ;

digit
  = [0-9] ;

letter
  = [A-Za-z] ;
